        private void btn1_Click(object sender, EventArgs e)
        {
            txtboxA.Text = txtboxA.Text + "1";
        }

        private void btnAC_Click(object sender, EventArgs e)
        {
            txtboxA.Clear();
        }

        private void btn2_Click(object sender, EventArgs e)
        {
            txtboxA.Text = txtboxA.Text + "2";
        }

        private void btn3_Click(object sender, EventArgs e)
        {
            txtboxA.Text = txtboxA.Text + "3";
        }

        private void btn0_Click(object sender, EventArgs e)
        {
            txtboxA.Text = txtboxA.Text + "0";
        }

        private void btnPunto_Click(object sender, EventArgs e)
        {
            txtboxA.Text = txtboxA.Text + ".";
        }

        private void btnMenos_Click(object sender, EventArgs e)
        {
            n1 = double.Parse(txtboxA.Text);
            operador = "-";

            txtboxA.Clear();
        }

        private void btnMas_Click(object sender, EventArgs e)
        {
            n1 = double.Parse(txtboxA.Text);
            operador = "+";

            txtboxA.Clear();
        }

        private void btnIgual_Click(object sender, EventArgs e)
        {
            n2 = double.Parse(txtboxA.Text);

            switch (operador)
            {
                case "-":
                    resta = n1 - n2;
                    txtboxA.Text = resta.ToString();
                    break;

                case "+":
                    suma = n1 + n2;
                    txtboxA.Text = suma.ToString();
                    break;

                default:
                    txtboxA.Text = n1.ToString();
                    break;
            }
        }
